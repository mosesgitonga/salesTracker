function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss   = SpreadsheetApp.openById(data.sheetId);

    var sheet = null;
    var sheets = ss.getSheets();
    for (var i = 0; i < sheets.length; i++) {
      if (String(sheets[i].getSheetId()) === String(data.gid)) {
        sheet = sheets[i]; break;
      }
    }
    if (!sheet) return respond({status:'error', message:'No tab found with gid '+data.gid});

    // Recalc-only action (no rows to write)
    if (data.action === 'recalc') {
      var totals = recalcTotals(sheet);
      return respond({status:'ok', totalCommission:totals.commission, totalNet:totals.net});
    }

    if (sheet.getLastRow() === 0) {
      sheet.appendRow(['DATE','USERNAME','FORM OF PAYMENT','GROSS PRICE','NET PRICE','COMMISSION 20%','NOTES']);
    }

    // If action is recalc, just update totals and return
    if (data.action === 'recalc') {
      var totals = recalcTotals(sheet);
      return respond({status:'ok', totalCommission:totals.commission, totalNet:totals.net});
    }

    var addedCount = 0, updatedCount = 0;

    var allValues = sheet.getDataRange().getValues();
    var lastNonEmpty = 0;
    for (var rv = 0; rv < allValues.length; rv++) {
      var isEmpty = allValues[rv].every(function(c){ return c===''||c===null; });
      if (!isEmpty) lastNonEmpty = rv + 1;
    }
    var appendStartRow = lastNonEmpty + 3 + 1;

    data.rows.forEach(function(r) {
      var rowData = [r.date,r.username,r.payment,r.gross,r.net,r.commission,r.notes];
      var rowUpdated = false;
      var lastRowIndex = sheet.getLastRow();
      for (var row = 5; row <= lastRowIndex; row++) {
        var dateCell = sheet.getRange(row,1).getValue().toString().trim();
        if (!dateCell) {
          sheet.getRange(row,1,1,rowData.length).setValues([rowData]);
          updatedCount++; rowUpdated = true; break;
        }
      }
      if (!rowUpdated) {
        sheet.getRange(appendStartRow+addedCount,1,1,rowData.length).setValues([rowData]);
        addedCount++;
      }
    });

    var totals = recalcTotals(sheet);
    return respond({status:'ok', added:addedCount, updated:updatedCount,
      totalProcessed:addedCount+updatedCount,
      totalCommission:totals.commission, totalNet:totals.net, tabName:sheet.getName()});

  } catch(err) { return respond({status:'error', message:err.toString()}); }
}

function doGet(e) {
  try {
    var action  = e.parameter.action;
    var sheetId = e.parameter.sheetId;
    var gid     = e.parameter.gid;

    if (action === 'recalc' && sheetId && gid) {
      var ss = SpreadsheetApp.openById(sheetId);
      var sheet = null;
      var sheets = ss.getSheets();
      for (var i = 0; i < sheets.length; i++) {
        if (String(sheets[i].getSheetId()) === String(gid)) { sheet = sheets[i]; break; }
      }
      if (!sheet) return respond({status:'error', message:'No tab found with gid '+gid});
      var totals = recalcTotals(sheet);
      return respond({status:'ok',
        totalCommission: Number(totals.commission),
        totalNet: Number(totals.net),
        tabName: sheet.getName()});
    }

    return respond({status:'ok', message:'Script is live'});
  } catch(err) { return respond({status:'error', message:err.toString()}); }
}

function recalcTotals(sheet) {
  var lastRow = sheet.getLastRow();
  var totalCommission = 0, totalNet = 0;
  if (lastRow >= 5) {
    var vals = sheet.getRange(5, 5, lastRow - 4, 2).getValues();
    for (var i = 0; i < vals.length; i++) {
      // Values may be numbers or strings like "8.00" — parse both safely
      var n = parseFloat(String(vals[i][0]).replace('$',''));
      var c = parseFloat(String(vals[i][1]).replace('$',''));
      if (!isNaN(n)) totalNet        += n;
      if (!isNaN(c)) totalCommission += c;
    }
  }
  totalCommission = Math.round(totalCommission * 100) / 100;
  totalNet        = Math.round(totalNet        * 100) / 100;
  sheet.getRange(3, 8).setValue(totalCommission); // H3 = Total Commission
  sheet.getRange(3, 9).setValue(totalNet);         // I3 = Total Net
  return {commission: totalCommission, net: totalNet};
}

function respond(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
