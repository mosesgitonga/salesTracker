Apps Script Setup

1. Open your Google Sheet
2. Click Extensions → Apps Script
3. Delete all existing code and paste the script below
4. Click Deploy → New Deployment → Web App
5. Set Execute as: Me and Who has access: Anyone
6. Copy the Web App URL and paste it into the field above 
function doPost(e) {
  try {
    var data   = JSON.parse(e.postData.contents);
    var ss     = SpreadsheetApp.openById(data.sheetId);
    var sheets = ss.getSheets();
    var sheet  = null;

    for (var i = 0; i < sheets.length; i++) {
      if (String(sheets[i].getSheetId()) === String(data.gid)) {
        sheet = sheets[i];
        break;
      }
    }

    if (!sheet) {
      return respond({status: 'error', message: 'No tab found with gid ' + data.gid});
    }

    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        'DATE','USERNAME','FORM OF PAYMENT',
        'GROSS PRICE','NET PRICE','COMMISSION 20%','NOTES'
      ]);
    }

    data.rows.forEach(function(r) {
      sheet.appendRow([
        r.date, r.username, r.payment,
        r.gross, r.net, r.commission, r.notes
      ]);
    });

    return respond({status: 'ok', added: data.rows.length, tabName: sheet.getName()});

  } catch(err) {
    return respond({status: 'error', message: err.toString()});
  }
}

function doGet(e) {
  return respond({status: 'ok', message: 'Script is live'});
}

function respond(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
⚠ Critical deployment step:
After pasting the script, go to Deploy → Manage Deployments,
click the pencil ✏ icon, change version to "New version", then hit Deploy.
If you skip this, the old script keeps running.
